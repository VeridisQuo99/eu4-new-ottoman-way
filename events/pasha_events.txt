# EVENT 1: The Senate Confirmation Event (Triggered on election)
country_event = {
    id = pasha_events.1
    title = "pasha_events.1.t"
    desc = "pasha_events.1.d"
    picture = PARLIAMENT_eventPicture

    is_triggered_only = yes

    # Seçilen adayın yeteneğine göre arka planda hizip güçlendirme
    immediate = {
		# Ilmiye (ADM Skill)
		if = { limit = { adm = 4 } add_faction_influence = { faction = ilmiye_faction influence = 10 } }
		if = { limit = { adm = 5 } add_faction_influence = { faction = ilmiye_faction influence = 10 } }
		if = { limit = { adm = 6 } add_faction_influence = { faction = ilmiye_faction influence = 10 } }

		# Kalemiye (DIP Skill)
		if = { limit = { dip = 4 } add_faction_influence = { faction = kalemiye_faction influence = 10 } }
		if = { limit = { dip = 5 } add_faction_influence = { faction = kalemiye_faction influence = 10 } }
		if = { limit = { dip = 6 } add_faction_influence = { faction = kalemiye_faction influence = 10 } }

		# Seyfiye (MIL Skill)
		if = { limit = { mil = 4 } add_faction_influence = { faction = seyfiye_faction influence = 10 } }
		if = { limit = { mil = 5 } add_faction_influence = { faction = seyfiye_faction influence = 10 } }
		if = { limit = { mil = 6 } add_faction_influence = { faction = seyfiye_faction influence = 10 } }
    }

    # Option A: Bribe the Senate
    option = {
        name = "pasha_events.1.a"
        add_years_of_income = -0.25
        add_corruption = 0.5
        add_government_power = {
            mechanic_type = pasha_power_mechanic
            power_type = pasha_power
            value = -10
        }
        custom_tooltip = "The Senate has been bribed to accept the new Pasha."

        ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				is_in_deficit = yes
			}
			modifier = {
				factor = 0
				corruption = 5
			}
			modifier = {
				factor = 0
				NOT = {
					has_government_power = {
						mechanic_type = pasha_power_mechanic
						power_type = pasha_power
						value = 20
					}
				}
			}
		}
    }

    # Option B: Concede to the Senate
    option = {
        name = "pasha_events.1.b"
        add_prestige = -10
        add_government_power = {
            mechanic_type = pasha_power_mechanic
            power_type = pasha_power
            value = -20
        }

        ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = {
					has_government_power = {
						mechanic_type = pasha_power_mechanic
						power_type = pasha_power
						value = 30
					}
				}
			}
		}
    }

    # Option C: Bypass the Senate
    option = {
        name = "pasha_events.1.c" 
        add_stability = -1
        add_republican_tradition = -5
        add_government_power = {
            mechanic_type = pasha_power_mechanic
            power_type = pasha_power
            value = 10
        }
        add_country_modifier = {
            name = "angry_senate_modifier"
            duration = 1460
        }

        ai_chance = {
			factor = 10 # AI generally avoids losing stability
			modifier = {
				factor = 0
				NOT = { stability = 1 } # AI will NEVER pick this if stability is 0 or negative
			}
			modifier = {
				factor = 3
				stability = 3 # If stability is maxed, AI is more willing to burn it
			}
            modifier = {
				factor = 0
				NOT = { republican_tradition = 50 } # Will not bypass senate if RT is below 50
			}
		}
    }
}

# EVENT 2: The 1444 Setup Event: Fate of the Sultanate (Triggered on startup)
country_event = {
    id = pasha_events.2
    title = "pasha_events.2.t"
    desc = "pasha_events.2.d"
    picture = COURT_eventPicture

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            set_country_flag = pasha_system_initialized
            change_government = republic
            set_variable = {
                which = sultan_count_var
                value = 0
            }
            define_ruler = {
                name = "Halil"
                dynasty = "Candarli"
                adm = 4
                dip = 4
                mil = 3
                age = 50
            }
            add_republican_tradition = 20
            add_government_reform = pasha_council_republic
        }
    }

    # Option A: Historical Path
    option = {
        name = "pasha_events.2.a" # Restore the Sultan's Power

		hidden_effect = {
			set_global_flag = pasha_sultan_mehmed_ii
            set_country_flag = historical_sultans_path
			
            change_variable = {
                which = sultan_count_var
                value = 1
            }
		}

        add_country_modifier = {
            name = sultan_mehmed_ii_modifier
            duration = -1
        }

		add_government_power = {
			mechanic_type = pasha_power_mechanic
			power_type = pasha_power
			value = 20 # Güç Sultanda toplanır (Artı değer)
		}

        ai_chance = {
			factor = 90 # AI strongly prefers the historical Sultan path
		}
    }

    # Option B: Random Path
    option = {
        name = "pasha_events.2.b"
        set_country_flag = random_sultans_path
        random_list = {
            33 = { add_country_modifier = { name = "random_sultan_militarist" duration = -1 } }
            33 = { add_country_modifier = { name = "random_sultan_diplomat" duration = -1 } }
            34 = { add_country_modifier = { name = "random_sultan_administrator" duration = -1 } }
        }

		add_government_power = {
			mechanic_type = pasha_power_mechanic
			power_type = pasha_power
			value = -20 # Güç Paşalara/Divana kayar (Eksi değer)
		}

        ai_chance = {
			factor = 10 # Rare alternate history path for AI
		}
    }
}

# EVENT 3: The Outbreak of the Crisis (Disaster triggered)
country_event = {
	id = pasha_events.3
	title = "pasha_events.3.t"
	desc = "pasha_events.3.d"
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes

	# Option A: Side with the Sultan
	option = {
		name = "pasha_events.3.a"
		
		# If the Palace has the balance of power (Value >= 50)
		if = {
			limit = {
				has_government_power = {
                    mechanic_type = pasha_power_mechanic
                    power_type = pasha_power
                    value = 50
                }
			}
			add_stability = -1
			capital_scope = { spawn_rebels = { type = noble_rebels size = 1 } }
		}
		
		# If the Parliament has the balance of power (Value < 50)
		if = {
			limit = {
				NOT = { 
                    has_government_power = {
                        mechanic_type = pasha_power_mechanic
                        power_type = pasha_power
                        value = 50
                    } 
                }
			}
			add_stability = -3
			capital_scope = { spawn_rebels = { type = noble_rebels size = 3 } }
		}
		
		set_country_flag = disaster_sided_with_sultan

        ai_chance = {
			factor = 25 # AI defaults to Sultan if status quo is not favorable
		}
	}

	# Option B: Maintain the Status Quo
	option = {
		name = "pasha_events.3.b"
		
		add_stability = -2
		add_country_modifier = {
			name = "political_gridlock"
			duration = 1800
		}
		
		set_country_flag = disaster_sided_with_status_quo

        ai_chance = {
			factor = 70 # AI heavily prefers maintaining the status quo and stability
		}
	}

	# Option C: Side with the Pasha
	option = {
		name = "pasha_events.3.c"
		
		# If the Palace has the balance of power (Value >= 50)
		if = {
			limit = {
				has_government_power = {
                    mechanic_type = pasha_power_mechanic
                    power_type = pasha_power
                    value = 50
                } 
			}
			add_stability = -3
			capital_scope = { spawn_rebels = { type = pretender_rebels size = 3 } }
		}
		
		# If the Parliament has the balance of power (Value < 50)
		if = {
			limit = {
				NOT = { 
                    has_government_power = {
                        mechanic_type = pasha_power_mechanic
                        power_type = pasha_power
                        value = 50
                    } 
                }
			}
			add_stability = -1
			capital_scope = { spawn_rebels = { type = pretender_rebels size = 1 } }
		}
		
		set_country_flag = disaster_sided_with_pasha

        ai_chance = {
			factor = 5 # AI almost never supports a full bureaucratic takeover
		}
	}
}

# EVENT 4: Death of the Background Sultan (Historical)
# This handles the transition from Mehmed II to Bayezid II
country_event = {
    id = pasha_events.4
    title = "pasha_events.4.t"
    desc = "pasha_events.4.d"
    picture = DEATH_OF_A_HEIR_eventPicture

    trigger = {
        tag = TUR
        has_country_flag = historical_sultans_path
		has_reform = pasha_council_republic
        is_year = 1481 # The historical death year of Mehmed II
        has_global_flag = pasha_sultan_mehmed_ii
		NOT = { has_global_flag = pasha_sultan_bayezid_ii }    
    }

    mean_time_to_happen = {
        months = 6
    }

    option = {
        name = "pasha_events.4.a" # Bayezid II ascends to the throne

        hidden_effect = {
			clr_global_flag = pasha_sultan_mehmed_ii
			set_global_flag = pasha_sultan_bayezid_ii

            change_variable = {
                which = sultan_count_var
                value = 1
            }
		}
		add_stability = 1
		add_government_power = {
			mechanic_type = pasha_power_mechanic
			power_type = pasha_power
			value = -10 # II. Bayezid dönemi güç merkeze göre biraz Paşalara kayar
		}

        remove_country_modifier = sultan_mehmed_ii_modifier

        add_country_modifier = {
            name = "sultan_bayezid_ii_modifier"
            duration = -1
        }
    }
}

# EVENT 5: Death of a Random Sultan (Random Path)
# Periodic check to see if the background Sultan dies
country_event = {
    id = pasha_events.5
    title = "pasha_events.5.t"
    desc = "pasha_events.5.d"
    picture = DEATH_OF_A_HEIR_eventPicture

    trigger = {
        has_country_flag = random_sultans_path
        OR = {
            has_country_modifier = random_sultan_militarist
            has_country_modifier = random_sultan_diplomat
            has_country_modifier = random_sultan_administrator
        }
    }

    mean_time_to_happen = {
        months = 360 # Average 30-year reign
    }

    immediate = {
        if = { limit = { has_country_modifier = random_sultan_militarist } remove_country_modifier = random_sultan_militarist }
        if = { limit = { has_country_modifier = random_sultan_diplomat } remove_country_modifier = random_sultan_diplomat }
        if = { limit = { has_country_modifier = random_sultan_administrator } remove_country_modifier = random_sultan_administrator }
    }

    # Option A: The Divan appoints a compliant candidate of their choosing
    option = {
        name = "pasha_events.5.a"
        add_country_modifier = { name = "random_sultan_administrator" duration = -1 }
        add_government_power = {
            mechanic_type = pasha_power_mechanic
            power_type = pasha_power
            value = -10
        }
        add_corruption = 1

        ai_chance = {
            factor = 40
            modifier = {
                factor = 0
                corruption = 7
            }
            modifier = {
                factor = 0
                NOT = {
                    has_government_power = {
                        mechanic_type = pasha_power_mechanic
                        power_type = pasha_power
                        value = 20
                    }
                }
            }
        }
    }

    # Option B: Let fate determine the successor's character
    option = {
        name = "pasha_events.5.b"
        random_list = {
            33 = { add_country_modifier = { name = "random_sultan_militarist" duration = -1 } }
            33 = { add_country_modifier = { name = "random_sultan_diplomat" duration = -1 } }
            34 = { add_country_modifier = { name = "random_sultan_administrator" duration = -1 } }
        }
        add_prestige = -5

        ai_chance = {
            factor = 40
        }
    }

    # Option C: Back the strongest claimant — a capable Sultan, whatever the cost
    option = {
        name = "pasha_events.5.c"
        add_country_modifier = { name = "random_sultan_militarist" duration = -1 }
        add_government_power = {
            mechanic_type = pasha_power_mechanic
            power_type = pasha_power
            value = 10
        }
        add_army_tradition = 5
        add_stability = -1

        ai_chance = {
            factor = 20
            modifier = {
                factor = 0
                NOT = { stability = 1 }
            }
            modifier = {
                factor = 3
                stability = 3
            }
        }
    }
}
