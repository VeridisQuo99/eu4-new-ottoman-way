namespace = pasha_events

# EVENT 1: The Senate Confirmation Event (Triggered on election)
country_event = {
    id = pasha_events.1
    title = "pasha_events.1.t"
    desc = "pasha_events.1.d"
    picture = PARLIAMENT_eventPicture

    is_triggered_only = yes

    trigger = {
        has_reform = pasha_council_republic
    }

    # Option A: Bribe the Senate (Shifts power to Parliament)
    option = {
        name = "pasha_events.1.a" 
        add_years_of_income = -0.25 
        add_corruption = 1.0
        change_variable = { which = palace_power_balance value = -5 }
        custom_tooltip = "The Senate has been bribed to accept the new Pasha."
    }

    # Option B: Concede to the Senate (Shifts power heavily to Parliament)
    option = {
        name = "pasha_events.1.b" 
        add_prestige = -5
        change_variable = { which = palace_power_balance value = -10 }
        every_owned_province = {
            limit = {
                has_seat_in_parliament = yes
            }
            add_local_autonomy = 10
        }
    }

    # Option C: Bypass the Senate (Shifts power to the Palace)
    option = {
        name = "pasha_events.1.c" 
        add_stability = -1
        add_republican_tradition = -5
        change_variable = { which = palace_power_balance value = 15 }
        add_country_modifier = {
            name = "angry_senate_modifier"
            duration = 1460
        }
    }
}

# EVENT 2: The 1444 Setup Event: Fate of the Sultanate (Triggered on startup)
country_event = {
    id = pasha_events.2
    title = "pasha_events.2.t"
    desc = "pasha_events.2.d"
    picture = COURT_eventPicture

    trigger = {
        tag = TUR
        is_year = 1444
        NOT = { has_country_flag = pasha_system_initialized }
    }

    immediate = {
        hidden_effect = {
            set_country_flag = pasha_system_initialized
            change_government = republic
            add_government_reform = pasha_council_republic
            set_variable = { which = palace_power_balance value = 50 }
            define_ruler = {
                name = "Halil"
                dynasty = "Candarli"
                adm = 4
                dip = 4
                mil = 3
                age = 50
            }
        }
    }

    # Option A: Historical Path
    option = {
        name = "pasha_events.2.a" 
        set_country_flag = historical_sultans_path
        add_country_modifier = {
            name = "sultan_mehmed_ii_modifier"
            duration = -1 
        }
    }

    # Option B: Random Path
    option = {
        name = "pasha_events.2.b" 
        set_country_flag = random_sultans_path
        random_list = {
            33 = { add_country_modifier = { name = "random_sultan_militarist" duration = -1 } }
            33 = { add_country_modifier = { name = "random_sultan_diplomat" duration = -1 } }
            34 = { add_country_modifier = { name = "random_sultan_administrator" duration = -1 } }
        }
    }
}

# EVENT 3: The Outbreak of the Crisis (Disaster triggered)
country_event = {
	id = pasha_events.3
	title = "pasha_events.3.t"
	desc = "pasha_events.3.d"
	picture = CIVIL_WAR_eventPicture

	is_triggered_only = yes

	# Option A: Side with the Sultan
	option = {
		name = "pasha_events.3.a"
		
		# If the Palace has the balance of power (Value >= 50)
		if = {
			limit = {
				check_variable = { which = palace_power_balance value = 50 }
			}
			add_stability = -1
			capital_scope = { spawn_rebels = { type = noble_rebels size = 1 } }
		}
		
		# If the Parliament has the balance of power (Value < 50)
		if = {
			limit = {
				NOT = { check_variable = { which = palace_power_balance value = 50 } }
			}
			add_stability = -3
			capital_scope = { spawn_rebels = { type = noble_rebels size = 3 } }
		}
		
		set_country_flag = disaster_sided_with_sultan
	}

	# Option B: Maintain the Status Quo
	option = {
		name = "pasha_events.3.b"
		
		add_stability = -2
		add_country_modifier = {
			name = "political_gridlock"
			duration = 3650
		}
		
		set_country_flag = disaster_sided_with_status_quo
	}

	# Option C: Side with the Pasha
	option = {
		name = "pasha_events.3.c"
		
		# If the Palace has the balance of power (Value >= 50)
		if = {
			limit = {
				check_variable = { which = palace_power_balance value = 50 }
			}
			add_stability = -3
			capital_scope = { spawn_rebels = { type = pretender_rebels size = 3 } }
		}
		
		# If the Parliament has the balance of power (Value < 50)
		if = {
			limit = {
				NOT = { check_variable = { which = palace_power_balance value = 50 } }
			}
			add_stability = -1
			capital_scope = { spawn_rebels = { type = pretender_rebels size = 1 } }
		}
		
		set_country_flag = disaster_sided_with_pasha
	}
}

# EVENT 4: Death of the Background Sultan (Historical)
# This handles the transition from Mehmed II to Bayezid II
country_event = {
    id = pasha_events.4
    title = "pasha_events.4.t"
    desc = "pasha_events.4.d"
    picture = DEATH_OF_A_HEIR_eventPicture

    trigger = {
        has_country_flag = historical_sultans_path
        has_country_modifier = sultan_mehmed_ii_modifier
        is_year = 1481 # The historical death year of Mehmed II
    }

    mean_time_to_happen = {
        months = 6
    }

    option = {
        name = "pasha_events.4.a" # Bayezid II ascends to the throne
        remove_country_modifier = sultan_mehmed_ii_modifier
        add_country_modifier = {
            name = "sultan_bayezid_ii_modifier"
            duration = -1
        }
    }
}

# EVENT 5: Death of a Random Sultan (Random Path)
# Periodic check to see if the background Sultan dies
country_event = {
    id = pasha_events.5
    title = "pasha_events.5.t"
    desc = "pasha_events.5.d"
    picture = DEATH_OF_A_HEIR_eventPicture

    trigger = {
        has_country_flag = random_sultans_path
        # Ensures there's a Sultan to die
        OR = {
            has_country_modifier = random_sultan_militarist
            has_country_modifier = random_sultan_diplomat
            has_country_modifier = random_sultan_administrator
        }
    }

    mean_time_to_happen = {
        months = 360 # Average 30-year reign
    }

    option = {
        name = "pasha_events.5.a" # Long live the new Sultan!
        # Clear old modifiers
        remove_country_modifier = random_sultan_militarist
        remove_country_modifier = random_sultan_diplomat
        remove_country_modifier = random_sultan_administrator
        
        # Assign a new random personality
        random_list = {
            33 = { add_country_modifier = { name = "random_sultan_militarist" duration = -1 } }
            33 = { add_country_modifier = { name = "random_sultan_diplomat" duration = -1 } }
            34 = { add_country_modifier = { name = "random_sultan_administrator" duration = -1 } }
        }
    }
}

# EVENT 6: The Sultan Restores the Monarchy
country_event = {
    id = pasha_events.6
    title = "pasha_events.6.t"
    desc = "pasha_events.6.d"
    picture = COURT_eventPicture

    trigger = {
        has_disaster = crisis_of_the_state
        has_country_flag = disaster_sided_with_sultan
        NOT = { num_of_rebel_armies = 1 }
        NOT = { has_country_flag = crisis_of_the_state_resolved }
    }

    mean_time_to_happen = {
        months = 1
    }

    option = {
        name = "pasha_events.6.a" # Long live the Sultan!
        set_country_flag = crisis_of_the_state_resolved
        change_government = monarchy
        add_government_reform = ottoman_government
        
        # Kills the active Pasha to allow the background Sultan's dynasty to take over
        kill_ruler = yes 
        add_stability = 1
    }
}

# EVENT 9: Power Struggle in the Capital (Periodic dynamic shifts)
country_event = {
    id = pasha_events.9
    title = "pasha_events.9.t"
    desc = "pasha_events.9.d"
    picture = COURT_eventPicture

    trigger = {
        has_reform = pasha_council_republic
        has_country_flag = pasha_system_initialized
    }

    # Triggers roughly every 10 years to keep the balance shifting
    mean_time_to_happen = {
        months = 120 
    }

    # Option A: The Sultan exerts influence
    option = {
        name = "pasha_events.9.a"
        change_variable = { which = palace_power_balance value = 10 }
        add_prestige = 5
        add_republican_tradition = -2
    }

    # Option B: The Pashas solidify their grip
    option = {
        name = "pasha_events.9.b"
        change_variable = { which = palace_power_balance value = -10 }
        add_republican_tradition = 2
        add_prestige = -5
    }
}
