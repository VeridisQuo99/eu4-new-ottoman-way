# Issue 1: Favouring the Men of the Sword (Seyfiye)
issue_military_campaign_funding = {
	category = 1
	
	# Only the Pasha Republic can see and debate this issue
	allow = {
		has_government_reform = pasha_council_republic
	}

	# What happens immediately when the issue is successfully passed
	effect = {
		# Increases Seyfiye influence by 15
		add_faction_influence = {
			faction = seyfiye_faction
			influence = 15
		}
		add_army_tradition = 10
	}

	# The passive bonus the country receives while this act is in effect (10 years)
	modifier = {
		global_manpower_modifier = 0.10
		land_maintenance_modifier = -0.05
	}
}
