# SLOT 6 — THE PORTE ASCENDANT (Pasha path)
pasha_porte_ascendant_missions = {
    slot = 6
    generic = no
    ai = yes

    potential = {
        tag = TUR
        has_reform = pasha_council_republic
    }

    has_country_shield = yes

    pasha_mission_divan_authority = {
        icon = mission_have_manufactories
        required_missions = { }
        position = 1

        trigger = {
            NOT = {
                has_government_power = {
                    mechanic_type = pasha_power_mechanic
                    power_type = pasha_power
                    value = 31
                }
            }
        }

        effect = {
            add_adm_power = 25
            add_prestige = 10
        }
    }

    pasha_mission_iltizam_reform = {
        icon = mission_build_up_to_force_limit
        required_missions = { pasha_mission_divan_authority }
        position = 2

        trigger = {
            NOT = {
                has_government_power = {
                    mechanic_type = pasha_power_mechanic
                    power_type = pasha_power
                    value = 21
                }
            }
            NOT = { corruption = 3 }
        }

        effect = {
            add_treasury = 50
            add_corruption = -0.5
        }
    }

    pasha_mission_grand_vizier_supreme = {
        icon = mission_empire
        required_missions = { pasha_mission_iltizam_reform }
        position = 3

        trigger = {
            has_country_flag = pasha_milestone_divan_ascendant
        }

        effect = {
            add_stability = 1
            add_republican_tradition = 10
        }
    }

    pasha_mission_age_of_pashas = {
        icon = mission_alliances
        required_missions = { pasha_mission_grand_vizier_supreme }
        position = 4

        trigger = {
            NOT = {
                has_government_power = {
                    mechanic_type = pasha_power_mechanic
                    power_type = pasha_power
                    value = 11
                }
            }
            faction_influence = { faction = seyfiye_faction influence = 20 }
            faction_influence = { faction = ilmiye_faction influence = 20 }
            faction_influence = { faction = kalemiye_faction influence = 20 }
        }

        effect = {
            add_adm_power = 50
            add_dip_power = 50
            add_country_modifier = {
                name = pasha_age_of_pashas_reward
                duration = -1
            }
        }
    }
}
