crisis_of_the_state = {
	potential = {
		has_reform = pasha_council_republic
	}

	can_start = {
		NOT = { republican_tradition = 35 }
	}

	can_stop = {
		republican_tradition = 80
	}

	progress = {
		modifier = {
			factor = 1
			NOT = { republican_tradition = 55 }
		}
		modifier = {
			factor = 1
			NOT = { republican_tradition = 45}
		}
		modifier = {
			factor = 1
			NOT = { republican_tradition = 35 }
		}
		modifier = {
			factor = 1
			NOT = { republican_tradition = 25 } 
		}
		modifier = {
			factor = 1
			NOT = { republican_tradition = 15}
		}
	}

	can_end = {
		NOT = { num_of_rebel_armies = 1 }

		custom_trigger_tooltip = {
			tooltip = crisis_resolved_tooltip
			has_country_flag = crisis_of_the_state_resolved
		}
		
		if = {
			limit = {
				has_country_flag = disaster_sided_with_status_quo
			}
			custom_trigger_tooltip = {
				tooltip = crisis_sided_with_divan
				NOT = { has_country_modifier = political_gridlock }	
				republican_tradition = 50		
			}
		}

		if = {
			limit = {
				OR = {
					has_country_flag = disaster_sided_with_pasha
					has_country_flag = disaster_sided_with_sultan
				}
			}
			custom_trigger_tooltip = {
				tooltip = crisis_sided_with_pasha_or_padisah
				stability = 1
			}
		}			
	}

	modifier = {
		global_unrest = 3
		stability_cost_modifier = 0.50
		global_tax_modifier = -0.15
	}

	on_start = pasha_events.3

	on_end = {
		clr_country_flag = crisis_of_the_state_resolved
	}
}
