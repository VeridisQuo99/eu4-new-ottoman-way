pasha_power_mechanic = {
    alert_icon_gfx = GFX_alerticons_government_mechanics
    alert_icon_index = 10

    powers = {
        pasha_power = {
            gui = pasha_power_window
            min = -50
            max = 50
            default = 0
            reset_on_new_ruler = no
            
            # Paşa Tarafı (Negatif: -1'den -50'ye doğru artar)
            scaled_modifier = {
                modifier = {
                    technology_cost = -0.1
                    development_cost = -0.1
                    max_absolutism = -15 
                }
                trigger = {
                    NOT = {
                        # Eğer güç 0 veya 0'dan büyükse bu şart DOĞRU olur, NOT bunu YANLIŞ yapar.
                        # Eğer güç -1, -10 veya -50 ise bu şart YANLIŞ olur, NOT bunu DOĞRU yapar ve modifier çalışır.
                        has_government_power = { 
                            mechanic_type = pasha_power_mechanic 
                            power_type = pasha_power 
                            value = 0 
                        }
                    }
                }
            }

            # Sultan Tarafı (Pozitif: 1'den 50'ye doğru artar)
            scaled_modifier = {
                modifier = {
                    global_unrest = -2.0
                    yearly_absolutism = 0.5
                    reform_progress_growth = -0.2 
                }
                trigger = {
                    # Sadece güç 1 veya daha büyükse çalışır. Tam 0 noktasında iki modifier da kapalı kalır.
                    has_government_power = { 
                        mechanic_type = pasha_power_mechanic 
                        power_type = pasha_power 
                        value = 1 
                    }
                }
            }
        }
    }

    interactions = {
        pasha_favor_sultan = {
            gui = "pasha_interaction_button_template"
            icon = GFX_russian_ability_ADM # Vanilla oyunda %100 çalışan ikon
            cost_type = republican_tradition
            cost = 10 
            trigger = {
                republican_tradition = 10
            }
            effect = {
                add_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = -10 }
                add_mil_power = -50 
            }
        }

        pasha_favor_pashas = {
            gui = "pasha_interaction_button_template"
            icon = GFX_russian_ability_DIP # Vanilla oyunda %100 çalışan ikon
            cost_type = adm_power
            cost = 50 
            effect = {
                add_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = 10 }
                add_corruption = 2 
                add_years_of_income = -0.5 
            }
        }
    }
}
