pasha_power_mechanic = {
    alert_icon_gfx = GFX_alerticons_government_mechanics
    alert_icon_index = 10

    powers = {
        pasha_power = {
            gui = pasha_power_window
            min = 0
            max = 100
            default = 50
            reset_on_new_ruler = no
        }
    }

    interactions = {
        # Left Button: Pulls power to 0 (Favor Pashas)
        pasha_favor_pashas = {
            gui = "pasha_interaction_button_template"
            icon = GFX_pasha_button_pashas_custom
            
            # Requires 50 Admin Power to be clicked
            trigger = {
                adm_power = 50
            }
            
            effect = {
                add_adm_power = -50 
                add_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = -10 }
                add_corruption = 1
                add_years_of_income = -0.5 
            }

            # pasha_favor_pashas butonu (Divanı güçlendirir)
            ai_chance = {
                factor = 0 # Normal şartlarda sıfırdır, statükoyu bozmaz.
                
                # Eğer güç barı 70'in üstüne çıkarsa AI dengelemek için basmaya başlar.
                modifier = {
                    factor = 50
                    has_government_power = {
                        mechanic_type = pasha_power_mechanic
                        power_type = pasha_power
                        value = 70
                    }
                }
                
                # Eğer güç barı 80'in (Disaster sınırı) üstüne çıkarsa AI panikler ve kesinlikle basar.
                modifier = {
                    factor = 200
                    has_government_power = {
                        mechanic_type = pasha_power_mechanic
                        power_type = pasha_power
                        value = 80
                    }
                }
            }
        }

        # Right Button: Pulls power to 100 (Favor Sultan)
        pasha_favor_sultan = {
            gui = "pasha_interaction_button_template"
            icon = GFX_pasha_button_sultan_custom
            
            # Requires 10 Republican Tradition to be clicked
            trigger = {
                republican_tradition = 5
            }
            
            effect = {
                add_republican_tradition = -5
                add_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = 10 }
            }


            # pasha_favor_sultan butonu (Sultanı güçlendirir)
            ai_chance = {
                factor = 0 # Normalde AI bu butona gereksiz yere basıp puan harcamaz.
                
                # Eğer güç barı 30'un altına inerse AI tehlikeyi sezer ve basmaya başlar.
                modifier = {
                    factor = 50
                    NOT = {
                        has_government_power = {
                            mechanic_type = pasha_power_mechanic
                            power_type = pasha_power
                            value = 30
                        }
                    }
                }
                
                # Eğer güç barı 20'nin (Disaster sınırı) altına inerse AI panikler ve kesinlikle basar.
                modifier = {
                    factor = 200
                    NOT = {
                        has_government_power = {
                            mechanic_type = pasha_power_mechanic
                            power_type = pasha_power
                            value = 20
                        }
                    }
                }

                modifier = {
                    factor = 0
                    NOT = { republican_tradition = 50 } # Prevents AI from clicking if it will compromise Republic stability
                }
            }
        }
    }
}
