pasha_power_mechanic = {
    alert_icon_gfx = GFX_alerticons_government_mechanics
    alert_icon_index = 10

    powers = {
        pasha_power = {
            gui = pasha_power_window
            min = -50
            max = 50
            default = 0
            reset_on_new_ruler = no
            
            # Paşa Tarafı (Negatif: 0'dan -50'ye)
            scaled_modifier = {
                modifier = {
                    technology_cost = -0.1
                    development_cost = -0.1
                    max_absolutism = -15 
                }
                trigger = {
                    NOT = {
                        has_government_power = { 
                            mechanic_type = pasha_power_mechanic 
                            power_type = pasha_power 
                            value = -0.1 # -0.1'den küçük (yani daha negatif) her değerde çalışır
                        }
                    }
                }
            }

            # Sultan Tarafı (Pozitif: 0'dan 50'ye)
            scaled_modifier = {
                modifier = {
                    global_unrest = -2.0
                    yearly_absolutism = 0.5
                    reform_progress_growth = -0.2 
                }
                trigger = {
                    has_government_power = { 
                        mechanic_type = pasha_power_mechanic 
                        power_type = pasha_power 
                        value = 0.1 # 0'dan büyük her değerde çalışır
                    }
                }
            }
        }
    }

    interactions = {
        # Favor the Sultan: Shifting power to authoritarian rule
        pasha_favor_sultan = {
            gui = "pasha_interaction_button_template" # Arayüz şablonuna bağlandı
            icon = GFX_pasha_button_sultan
            cost_type = republican_tradition
            cost = 10 # 10 RT is a massive, highly punishing cost in EU4
            trigger = {
                republican_tradition = 10
            }
            effect = {
                add_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = -10 }
                # Additional heavy penalty
                add_mil_power = -50 
            }
        }

        # Favor the Pashas: Bribing the council and giving concessions
        pasha_favor_pashas = {
            gui = "pasha_interaction_button_template" # Arayüz şablonuna bağlandı
            icon = GFX_pasha_button_pashas
            cost_type = adm_power
            cost = 50 
            # No trigger needed for adm_power as cost_type inherently checks it
            effect = {
                add_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = 10 }
                # Extremely punishing long-term economic and structural costs
                add_corruption = 2 
                add_years_of_income = -0.5 # Deducts 6 months of gross income
            }
        }
    }
}
