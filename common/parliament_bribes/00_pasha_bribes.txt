# 1. Seyfiye Rüşveti: Ulufe Dağıtımı
bribe_pasha_seyfiye = {
	trigger = {
		owner = { has_reform = pasha_council_republic }
	}
	effect = {
		owner = {
			add_years_of_income = -0.1 
			add_faction_influence = {
				faction = seyfiye_faction
				influence = 5
			}
		}
	}
	ai_will_do = {
		factor = 1000 
		modifier = {
			factor = 0
			owner = {
				faction_influence = {
					faction = seyfiye_faction
					influence = 70
				}
			}
		}
		modifier = {
			factor = 0
			owner = { is_in_deficit = yes }
		}
	}
}
