# 1. Seyfiye Rüşveti: Ulufe Dağıtımı
bribe_pasha_seyfiye = {
	trigger = {
		# Base scope is Country. Checked directly.
		has_reform = pasha_council_republic
	}
	effect = {
		# Country effects are applied directly
		add_years_of_income = -0.1 
		add_faction_influence = {
			faction = seyfiye_faction
			influence = 5
		}
	}
	ai_will_do = {
		factor = 1000 
		modifier = {
			factor = 0
			faction_influence = {
				faction = seyfiye_faction
				influence = 70
			}
		}
		modifier = {
			factor = 0
			is_in_deficit = yes
		}
	}
}

# 2. İlmiye Rüşveti: Vakıf Bağışla
bribe_pasha_ilmiye = {
	trigger = {
		has_reform = pasha_council_republic
	}
	effect = {
		# FROM targets the specific province (seat) receiving the bribe
		FROM = {
			add_local_autonomy = 5 
		}
		add_faction_influence = {
			faction = ilmiye_faction
			influence = 5
		}
	}
	ai_will_do = {
		factor = 1000
		modifier = {
			factor = 0
			faction_influence = {
				faction = ilmiye_faction
				influence = 70
			}
		}
	}
}
