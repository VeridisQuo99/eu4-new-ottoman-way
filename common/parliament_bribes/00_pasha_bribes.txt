# 1. Seyfiye Rüşveti: Ulufe Dağıtımı
bribe_pasha_seyfiye = {
	trigger = {
		FROM = { has_reform = pasha_council_republic }
	}
	effect = {
		FROM = {
			add_years_of_income = -0.1 # Hazineden yıllık gelirin %10'u kadar para çıkar
			add_faction_influence = {
				faction = seyfiye_faction
				influence = 5
			}
		}
	}
	ai_will_do = {
		factor = 10
		modifier = {
			factor = 0
			FROM.faction_influence = {
				faction = seyfiye_faction
				value = 70
			}
		}
		# Yapay zeka bütçe açığı veriyorsa ulufe dağıtmaz
		modifier = {
			factor = 0
			FROM = { is_in_deficit = yes }
		}
	}
}

# 2. İlmiye Rüşveti: Vakıf Bağışla
bribe_pasha_ilmiye = {
	trigger = {
		FROM = { has_reform = pasha_council_republic }
	}
	effect = {
		# Vakıflar vergiden muaf olduğu için o toprağın (ROOT) otonomisi artar
		add_local_autonomy = 5 
		
		FROM = {
			add_faction_influence = {
				faction = ilmiye_faction
				influence = 5
			}
		}
	}
	ai_will_do = {
		factor = 10
		modifier = {
			factor = 0
			FROM.faction_influence = {
				faction = ilmiye_faction
				value = 70
			}
		}
	}
}

# 3. Kalemiye Rüşveti: İltizam Hakkı Ver
bribe_pasha_kalemiye = {
	trigger = {
		FROM = { has_reform = pasha_council_republic }
	}
	effect = {
		
		FROM = {
			add_corruption = 0.5 # Devlet içi yozlaşmayı tetikler
			add_faction_influence = {
				faction = kalemiye_faction
				influence = 5
			}
		}
	}
	ai_will_do = {
		factor = 10
		modifier = {
			factor = 0
			FROM.faction_influence = {
				faction = kalemiye_faction
				value = 70
			}
		}
		# Yapay zeka corruption seviyesi zaten çok yüksekse bu rüşvetten kaçınır
		modifier = {
			factor = 0
			FROM = { corruption = 3 }
		}
	}
}
