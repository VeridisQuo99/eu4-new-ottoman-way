pasha_tier_1_absolute = {
    potential = {
        has_government_mechanic = pasha_power_mechanic
    }
    trigger = {
        # Yuvarlak top göstergesi 0 - 20 aralığındayken (Mutlak Paşa Gücü)
        NOT = { has_government_power = { mechanic_type = pasha_power_mechanic power_type = pasha_power value = 20 } }
    }
    technology_cost = -0.15
    development_cost = -0.15
    max_absolutism = -20
}
